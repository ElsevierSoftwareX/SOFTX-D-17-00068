147 MatrixXd EKF_AUS::ComputeHEf(FP NonLinH, MatrixXd Ef, double LinearScale, MatrixXd& xf)
148 {
149  MatrixXd hef, xfp(xf.rows(),1), hefcol;
150 
151  double invLinearScale = 1. / LinearScale;
152 
153  hef.resize( p, Ef.cols() );
154  hefcol.resize( p , 1 );
155 
156  for(long j=0; j<Ef.cols(); j++)
157  {
158  xfp.col(0) = xf.col(0) + LinearScale * Ef.col(j);
159  hefcol = invLinearScale * ( NonLinH(xfp) - NonLinH(xf) ) ;
160 
161  for(long i=0; i<p; i++) hef(i,j) = hefcol(i,0);
162  }
163  return hef;
164 }

178  MatrixXd& xf, MatrixXd& Xf)

194  alog << "---------------------------------------\n";

200

204 
205  alog << "Orthonormalizing.....\n\n" ; cout.flush();
206 
207  MatrixXd Ef = gramsh(Xf);
208 
209  MatrixXd ef = Xf.transpose() * Ef;
210  MatrixXd gamma1 = ef.transpose() *ef;
211  MatrixXd gamma2;
212 
213 
214  gamma2.resize(Xf.cols(),Xf.cols());
215 
216  alog << "Computing H*Ef...\n\n";
217 
218  MatrixXd HEf = ComputeHEf(NonLinH, Ef, 0.1, xf);
219  MatrixXd ToBeInv, Inv, check, I, CT;
220 
221  // ToBeInv = R + HEf * gamma1 * HEf.transpose();
222 
223  cout << HEf.rows() << " x " << HEf.cols() << endl;
224  cout << gamma1.rows() << " x " << gamma1.cols() << endl;
225 
226  MatrixXd tmp1, Kanom, tmp3, tmp2;
227 
228  ToBeInv = HEf * gamma1 * HEf.transpose();
229 
230  for(long i=0; i<ToBeInv.rows(); i++)
231  ToBeInv(i,i) += R(i,0);
232 
233  try{
234  Inv = ToBeInv.inverse(); // matrix inversion
235  }
236  catch(int inversione_flag){
237  cout << "EKF_AUS_NL::Assimilate: \nProblems in the matrix inversione, stop now, sorry.\n\n";
238  throw int(123);

241  /*

251

255 
256  // K = Ef * gamma1 * HEf.transpose() * Inv;
257  // Eq.(10) of Nonlin. Proc. Geophys. 18, 243-250 (2011).
258 
259  MatrixXd Ea;
260  MatrixXd xa;
261 
262  xa = xf + Kanom;
263

265 
266  alog << "Diagonalizing via singular values (more stable)....." << endl;
267 
268  JacobiSVD<Eigen::MatrixXd> svd(gamma2, ComputeThinU | ComputeThinV);
269 
270  cout << "The singular values are:" << endl << svd.singularValues() << endl;

273 
274  cout << endl;

277 
278  long int i,j;
279 
280 
281  for(i=0;i<Xf.cols();i++)
282  {
283  l(i) = sqrt( ev(i) ) * inflation + add_inflation;
284  }
285 
286  alog << "EIGEN: ";
287  for(i=0; i< Xf.cols(); i++) alog << l(i) << " ";
288  alog << endl;

293

295  for(j=0;j<n;j++)

360  {
361  outXf(l,k - HalfNumberNonLinPert() ) +=\
