249 MatrixXd EKF_AUS::ComputeHEf(FP NonLinH, MatrixXd Ef, double LinearScale, MatrixXd& xf)
250 {
251  MatrixXd hef, xfp(xf.rows(),1), hefcol;
252 
253  double invLinearScale = 1. / LinearScale;
254 
255  hef.resize( p, Ef.cols() );
256  hefcol.resize( p , 1 );
257 
258  for(long j=0; j<Ef.cols(); j++)
259  {
260  xfp.col(0) = xf.col(0) + LinearScale * Ef.col(j);
261  hefcol = invLinearScale * ( NonLinH(xfp) - NonLinH(xf) ) ;
262 
263  for(long i=0; i<p; i++) hef(i,j) = hefcol(i,0);
264  }
265  return hef;
