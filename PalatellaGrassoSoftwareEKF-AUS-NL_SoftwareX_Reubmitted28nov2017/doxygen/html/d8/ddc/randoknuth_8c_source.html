65  for (j=0;j<KK;j++) {
66  u[j]=ss; /* bootstrap the buffer */
67  ss+=ss; if (ss>=1.0) ss-=1.0-2*ulp; /* cyclic shift of 51 bits */
68  }
69  u[1]+=ulp; /* make u[1] (and only u[1]) "odd" */
70  for (s=seed&0x3fffffff,t=TT-1; t; ) {
71  for (j=KK-1;j>0;j--)
72  u[j+j]=u[j],u[j+j-1]=0.0; /* "square" */
73  for (j=KK+KK-2;j>=KK;j--) {

75  u[j-KK]=mod_sum(u[j-KK],u[j]);
76  }

78  for (j=KK;j>0;j--) u[j]=u[j-1];
79  u[0]=u[KK]; /* shift the buffer cyclically */
80  u[LL]=mod_sum(u[LL],u[KK]);
